RUN apt-get update && apt-get install -y \
    wget \
    ca-certificates \
    fonts-liberation \
    libasound2 \
    libatk-bridge2.0-0 \
    libatk1.0-0 \
    libcups2 \
    libdbus-1-3 \
    libgdk-pixbuf2.0-0 \
    libnspr4 \
    libnss3 \
    libx11-xcb1 \
    libxcomposite1 \
    libxdamage1 \
    libxrandr2 \
    xdg-utils \
    xvfb \
    fluxbox \
    novnc \
    websockify \
    chromium \
    --no-install-recommends && \
    rm -rf /var/lib/apt/lists/*

# Optional: install puppeteer globally if you want Node control, not required for GUI-only
RUN npm install -g puppeteer

EXPOSE 6080 9222

COPY start.sh /start.sh
RUN chmod +x /start.sh

CMD ["/start.sh"]
